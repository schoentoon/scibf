29 void irc_conn_readcb(struct bufferevent *bev, void* args) {
30  DEBUG(255, "irc_conn_readcb(%p, %p);", bev, args);
31  struct evbuffer* input = bufferevent_get_input(bev);
32  struct evbuffer* output = bufferevent_get_output(bev);
33  struct server* server = (struct server*) args;
34  size_t len;
35  char* line = evbuffer_readln(input, &len, EVBUFFER_EOL_CRLF);
36  while (line) {
37  DEBUG(254, "In: '%s'", line);
38  char buf[MAX_LINE];
39  static const char* IRC_PING_SSCANF = "PING %s";
40  static const char* IRC_PONG_PRINTF = "PONG %s\r\n";
41  if (sscanf(line, IRC_PING_SSCANF, buf) == 1)
42  evbuffer_add_printf(output, IRC_PONG_PRINTF, buf);
43  else if (len >= 3) {
44  static const char* IRC_NUMBERED_EVENT = "%s %d %[^\r\n]";
45  static const char* IRC_GENERAL_EVENT = "%s %s %[^\r\n]";
46  int uRaw = 0;
47  char rest[MAX_LINE];
48  char server_name[MAX_LINE];
49  if (sscanf(line, IRC_NUMBERED_EVENT, server_name, &uRaw, rest) == 3) {
50  switch (uRaw) {
51  case 352: { /* :localhost 352 IAmABot #test scibf 7499CEAE.E300844E.CC9601B0.IP localhost IAmABot H :0 scibf */
52  static const char* WHO_SSCANF = "%s %s %[^\r\n]";
53  char me[NICK_LEN];
54  char chan[CHAN_LEN];
55  char who_output[MAX_LINE];
56  if (sscanf(rest, WHO_SSCANF, me, chan, who_output) == 3) {
57  struct channel* channel = get_channel(server->conn, chan);
58  parse_who_header(channel, who_output);
59  };
60  break;
61  };
62  case 353: { /* :localhost 353 IAmABot = #test :IAmABot ~@Schoentoon @SomeOp */
63  static const char* NAMES_REST_SSCANF = "%s = %s :%[^\r\n]";
64  char me[NICK_LEN];
65  char chan[CHAN_LEN];
66  char names[MAX_LINE];
67  if (sscanf(rest, NAMES_REST_SSCANF, me, chan, names) == 3) {
68  struct channel* channel = get_channel(server->conn, chan);
69  fill_from_names(channel, names);
70  }
71  break;
72  };
73  case 366: { /* :localhost 366 IAmABot #test :End of /NAMES list. */
74  DEBUG(255, "rest: '%s'", rest);
75  static const char* END_OF_NAMES_CHAN = "%s %s";
76  char me[NICK_LEN];
77  char chan[CHAN_LEN];
78  if (sscanf(rest, END_OF_NAMES_CHAN, me, chan) == 2) {
79  static const char* WHO_CHAN_PRINTF = "WHO %s\r\n";
80  evbuffer_add_printf(output, WHO_CHAN_PRINTF, chan);
81  }
82  break;
83  };
84  case 422: /* :localhost 422 IAmABot :MOTD File is missing */
85  case 376: { /* :localhost 376 IAmABot :End of MOTD command. */
86  static const char* IRC_JOIN_PRINTF = "JOIN %s\r\n";
87  unsigned int i;
88  for (i = 0; server->channels[i]; i++)
89  evbuffer_add_printf(output, IRC_JOIN_PRINTF, server->channels[i]);
90  break;
91  };
92  };
93  } else {
94  char event[MAX_LINE];
95  if (sscanf(line, IRC_GENERAL_EVENT, server_name, event, rest) == 3) {
96  static const char* IRC_JOIN_EVENT = "JOIN";
97  static const char* IRC_NICK_EVENT = "NICK";
98  static const char* IRC_PART_EVENT = "PART";
99  static const char* IRC_QUIT_EVENT = "QUIT";
100  static const char* IRC_PRIVMSG_EVENT = "PRIVMSG";
101  if (strcmp(event, IRC_JOIN_EVENT) == 0) {
102  struct channel* channel = get_channel(server->conn, &rest[1]);
103  struct user* user = new_user(&server_name[1]);
104  add_user_to_channel(channel, user);
105  } else if (strcmp(event, IRC_NICK_EVENT) == 0) {
106  char buf[NICK_LEN];
107  if (get_nickname(server_name, buf)) {
108  struct channel* node = server->conn->channels;
109  while (node) {
110  struct user* user = get_user_from_channel(node, buf);
111  if (user) {
112  DEBUG(255, "Changing user '%s' to '%s' in channel %s", user->nick, &rest[1], node->name);
113  free(user->nick);
114  user->nick = strdup(&rest[1]);
115  }
116  node = node->next;
117  };
118  }
119  } else if (strcmp(event, IRC_PART_EVENT) == 0) {
120  struct channel* channel = get_channel(server->conn, &rest[1]);
121  if (channel) {
122  char buf[NICK_LEN];
123  if (get_nickname(server_name, buf)) {
124  struct user* user = get_user_from_channel(channel, buf);
125  remove_user_from_channel(channel, user);
126  }
127  }
128  } else if (strcmp(event, IRC_QUIT_EVENT) == 0) {
129  char buf[NICK_LEN];
130  if (get_nickname(server_name, buf)) {
131  struct channel* node = server->conn->channels;
132  while (node) {
133  struct user* user = get_user_from_channel(node, buf);
134  remove_user_from_channel(node, user);
135  node = node->next;
136  };
137  };
138  } else if (strcmp(event, IRC_PRIVMSG_EVENT) == 0) {
139  static const char* PARSE_CHAN_PRIVMSG = "%s :%[^\r\n]";
140  char chan[CHAN_LEN];
141  char buf[MAX_LINE];
142  if (sscanf(rest, PARSE_CHAN_PRIVMSG, chan, buf) == 2) {
143  struct channel* channel = get_channel(server->conn, chan);
144  if (channel) {
145  char nickname[NICK_LEN];
146  if (get_nickname(server_name, nickname)) {
147  struct user* user = get_user_from_channel(channel, nickname);
148  if (user) { /* Call the channel message callbacks from here. */
149  static const char* PARSE_ACTION = "\001ACTION %[^\001]";
150  char action[MAX_LINE];
151  if (sscanf(buf, PARSE_ACTION, action) == 1) {
152  DEBUG(1, "%s %s", user->nick, action);
153  } else
154  DEBUG(1, "<%s> %s", user->nick, buf);
155  }
156  }
157  }
158  }
159  }
160  }
161  }
162  }
163  free(line);
164  line = evbuffer_readln(input, &len, EVBUFFER_EOL_CRLF);
165  }
166 };

173 static void reconnectServer(evutil_socket_t fd, short event, void* args) {
174  DEBUG(255, "reconnectServer(%d, 0x%02x, %p);", fd, event, args);
175  struct reconnect_struct* reconnect_struct = (struct reconnect_struct*) args;
176  startConnection(reconnect_struct->server, reconnect_struct->base);
177  free(reconnect_struct);

180 void irc_conn_eventcb(struct bufferevent *bev, short event, void* args) {
181  DEBUG(255, "irc_conn_eventcb(%p, 0x%02x, %p);", bev, event, args);
182  if (!(event & BEV_EVENT_CONNECTED)) {
183  struct event_base* base = bufferevent_get_base(bev);
184  struct server* node = (struct server*) args;
185  free_connection(node->conn);
186  node->conn = NULL;
187  struct timeval tv = { node->retry_time, 0 };
188  struct reconnect_struct* reconnect_struct = malloc(sizeof(struct reconnect_struct));
189  reconnect_struct->server = node;
190  reconnect_struct->base = base;
191  event_base_once(base, -1, EV_TIMEOUT, reconnectServer, reconnect_struct, &tv);
192  }
193 };
