40 int parse_config(char* config_file) {
41  if (global_config)
42  return 0;
43  FILE *f = fopen(config_file, "r");
44  if (f == NULL) {
45  fprintf(stderr, "Error '%s' while opening '%s'.\n", strerror(errno), config_file);
46  return 0;
47  }
48  global_config = malloc(sizeof(struct config));
49  struct server* current_server = NULL;
50  char line_buffer[BUFSIZ];
51  unsigned int line_count = 0;
52  while (fgets(line_buffer, sizeof(line_buffer), f)) {
53  line_count++;
54  if (strlen(line_buffer) == 1 || line_buffer[0] == '#')
55  continue;
56  char key[BUFSIZ];
57  char value[BUFSIZ];
58  if (sscanf(line_buffer, "%[a-z_] = %[^\t\n]", key, value) == 2) {
59  DEBUG(255, "key: '%s', value: '%s'", key, value);
60  if (strcasecmp(key, "name") == 0) {
61  if (current_server) {
62  if (!current_server->address || !current_server->port
63  || !current_server->username || !current_server->nickname) {
64  fprintf(stderr, "Error, missing a required key for server %s\n", current_server->unique_name);
65  return 0;
66  }
67  }
68  current_server = getServer(value);
69  if (!current_server->unique_name)
70  current_server->unique_name = strdup(value);
71  } else if (current_server == NULL) {
72  fprintf(stderr, "You specified the key %s before you specified the name key, which is required.\n", key);
73  fprintf(stderr, "You should probably go and fix your config.\n");
74  return 0;
75  } else if (strcasecmp(key, "address") == 0) {
76  if (current_server->address)
77  free(current_server->address);
78  current_server->address = strdup(value);
79  } else if (strcasecmp(key, "port") == 0) {
80  long port = strtol(value, NULL, 10);
81  if ((errno == ERANGE || (port == LONG_MAX || port == LONG_MIN)) || (errno != 0 && port == 0) || port < 0 || port > 65535) {
82  fprintf(stderr, "Error at line %d. Port %ld is out of range.\n", line_count, port);
83  return 0;
84  } else
85  current_server->port = port;
86  } else if (strcasecmp(key, "username") == 0) {
87  if (current_server->username)
88  free(current_server->username);
89  current_server->username = strdup(value);
90  } else if (strcasecmp(key, "nickname") == 0) {
91  if (current_server->nickname)
92  free(current_server->nickname);
93  current_server->nickname = strdup(value);
94  } else if (strcasecmp(key, "password") == 0) {
95  if (current_server->password)
96  free(current_server->password);
97  current_server->password = strdup(value);
98  } else if (strcasecmp(key, "channels") == 0) {
99  char* token = strtok(value, ",");
100  while (token) {
101  if (token[0] == '#') {
102  unsigned int next_chan = 0;
103  if (!current_server->channels)
104  current_server->channels = malloc(sizeof(char*) * 2);
105  else {
106  while (current_server->channels[++next_chan]);
107  current_server->channels = realloc(current_server->channels, sizeof(char*) * (next_chan + 1));
108  }
109  current_server->channels[next_chan] = strdup(token);
110  current_server->channels[next_chan + 1] = NULL;
111  }
112  token = strtok(NULL, ",");
113  };
114  } else if (strcasecmp(key, "retry_time") == 0) {
115  errno = 0;
116  long tmp = strtol(value, NULL, 10);
117  if (errno == 0 && tmp > 0)
118  current_server->retry_time = (unsigned int) tmp;
119  } else if (strcasecmp(key, "timeout") == 0) {
120  errno = 0;
121  long tmp = strtol(value, NULL, 10);
122  if (errno == 0 && tmp > 0 && tmp < 256)
123  current_server->timeout = (unsigned char) tmp;
124  };
125  } else {
126  fprintf(stderr, "Parsing error at line %d.\n", line_count);
127  return 0;
128  }
129  };
130  if (current_server) {
131  if (!current_server->address || !current_server->port
132  || !current_server->username || !current_server->nickname) {
133  fprintf(stderr, "Error, missing a required key for server %s\n", current_server->unique_name);
134  return 0;
135  }
136  }
137  fclose(f);
138  return 1;
139 };

141 struct server* getServer(char* name) {
142  if (!name)
143  return NULL;
144  struct server* node = global_config->servers;
145  if (!node) {
146  global_config->servers = malloc(sizeof(struct server));
147  memset(global_config->servers, 0, sizeof(struct server));
148  global_config->servers->retry_time = 10; /* 10 seconds as default.. */
149  global_config->servers->timeout = 90;
150  return global_config->servers;
151  }
152  for (;;) {
153  if (strcmp(name, node->unique_name) == 0)
154  return node;
155  if (!node->next) /* Doing this check right here and breaking out of it */
156  break; /* so I won't have to loop through all of them again to get */
157  node = node->next; /* the last one to append a new one. */
158  }
159  node->next = malloc(sizeof(struct server));
160  memset(node->next, 0, sizeof(struct server));
161  node->next->retry_time = 10; /* 10 seconds as default.. */
162  node->next->timeout = 90;
163  return node->next;
164 };

166 int dispatch_config(struct event_base* base) {
167  struct server* node = global_config->servers;
168  if (!node)
169  return 0;
170  if (!dns)
171  dns = evdns_base_new(base, 1);
172  while (node) {
173  startConnection(node, base);
174  node = node->next;
175  };
176  return 1;
177 };

179 int startConnection(struct server* server, struct event_base* base) {
180  if (server->conn && server->conn->conn)
181  return 0;
182  if (!server->conn)
183  server->conn = malloc(sizeof(struct connection));
184  memset(server->conn, 0, sizeof(struct connection));
185  server->conn->conn = bufferevent_socket_new(base, -1, BEV_OPT_CLOSE_ON_FREE);
186  struct timeval timeout = {server->timeout, 0};
187  bufferevent_set_timeouts(server->conn->conn, &timeout, NULL);
188  bufferevent_socket_connect_hostname(server->conn->conn, dns, AF_INET, server->address, server->port);
189  bufferevent_setcb(server->conn->conn, irc_conn_readcb, NULL, irc_conn_eventcb, server);
190  bufferevent_enable(server->conn->conn, EV_READ);
191  struct evbuffer* output = bufferevent_get_output(server->conn->conn);
192  if (server->password)
193  evbuffer_add_printf(output, "PASS %s\r\n", server->password);
194  evbuffer_add_printf(output, "NICK %s\r\n", server->nickname);
195  evbuffer_add_printf(output, "USER %s \"%s\" \"%s\" :%s\r\n", server->username, server->username, server->username, server->username);
196  return 1;
197 };

199 int free_connection(struct connection* connection) {
200  if (!connection)
201  return 0;
202  if (connection->conn)
203  bufferevent_free(connection->conn);
204  if (connection->channels) {
205  struct channel* chan_node = connection->channels;
206  while (chan_node) {
207  struct channel* tmp_chan = chan_node;
208  chan_node = chan_node->next;
209  free_channel(tmp_chan);
210  };
211  };
212  free(connection->hostname);
213  free(connection->nickname);
214  free(connection->username);
215  free(connection);
216  return 1;
217 };
