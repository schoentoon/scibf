44 struct channel* get_channel(struct connection* connection, char* channel) {
45  if (!connection || !channel)
46  return NULL;
47  if (!connection->channels) {
48  connection->channels = new_channel(channel);
49  return connection->channels;
50  }
51  struct channel* node = connection->channels;
52  for (;;) {
53  if (strcmp(node->name, channel) == 0)
54  return node;

58  };
59  node->next = new_channel(channel);

63 int add_user_to_channel(struct channel* channel, struct user* user) {
64  if (!channel || !user)
65  return 0;
66  if (!channel->users) {
67  channel->users = user;
68  return 1;
69  }
70  struct user* node = channel->users;
71  for (;;) {
72  if (strcmp(node->nick, user->nick) == 0) {
73  if (user->mode == node->mode)

76  node->mode = user->mode;
77  return 0;

79  }
80  if (!node->next)
81  break;
82  node = node->next;
83  };
84  node->next = user;
85  return 1;

88 int fill_from_names(struct channel* channel, char* raw_names) {

94  struct user* user = new_user(name);

114 int remove_user_from_channel(struct channel* channel, struct user* user) {
115  if (!channel || !user)
116  return 0;
117  struct user* node = channel->users;
118  while (node) {
119  if (node->next == user) {
120  if (user->next)
121  node->next = user->next;
122  else
123  node->next = NULL;
124  free_user(user);
125  return 1;
126  }
127  node = node->next;
128  };
